CREATE TYPE gender_type AS ENUM ('male', 'female');
CREATE TYPE state_type_enum AS ENUM ('hypnotic', 'normal', 'interacting');


CREATE TABLE Person (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    gender gender_type NOT NULL,
    current_location VARCHAR(50) NOT NULL
);

CREATE TABLE Object (
    id BIGSERIAL PRIMARY KEY,
    type VARCHAR(50) NOT NULL,
    length NUMERIC(5,2) NOT NULL,
    physical_properties VARCHAR(100) NOT NULL
);

CREATE TABLE State (
    id BIGSERIAL PRIMARY KEY REFERENCES Person(id),
    state_type state_type_enum NOT NULL,
    person_id BIGINT NOT NULL REFERENCES Person(id)
);

CREATE TABLE ActionType (
    id BIGSERIAL PRIMARY KEY,
    action_name VARCHAR(50) NOT NULL
);

CREATE TABLE PersonAction (
    id BIGSERIAL PRIMARY KEY,
    person_id BIGINT NOT NULL REFERENCES Person(id),
    action_type_id BIGINT NOT NULL REFERENCES ActionType(id),
    action_start_time TIMESTAMP NOT NULL
);

CREATE TABLE ObjectInteraction (
    id BIGSERIAL PRIMARY KEY,
    person_id BIGINT NOT NULL REFERENCES Person(id),
    object_id BIGINT NOT NULL REFERENCES Object(id),
    result VARCHAR(50) NOT NULL
);
